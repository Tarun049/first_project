
/**  adding meta box*/

// add_action("add_meta_boxes", "adding_meta_box_fn");
function adding_meta_box_fn()
{
    add_meta_box("genre_meta_box", "Genre", "genre_callback_fn", "book", "", "default");
}

// callback function of metabox #genre
function genre_callback_fn()
{
    $custom_genre = get_terms(["taxonomy" => "genre", "hide_empty" => false]);
    foreach ($custom_genre as $term_data) {
    ?>
        <input type="checkbox" name="custom_genre_name" value="" id="custom_genre_name">
        <label for="custom_genre_name"><?php echo $term_data->name ?></label>
    <?php
        echo "<br>";
    }
}
/**  end of meta box*/
